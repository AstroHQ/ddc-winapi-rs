Implementation of DDC/CI traits on Windows.

Example

extern crate ddc;

use ddc::Ddc;
use ddc_winapi::Monitor;

for mut ddc in Monitor::enumerate().unwrap() {
    let mccs_version = ddc.get_vcp_feature(0xdf).unwrap();
    println!("MCCS version: {:04x}", mccs_version.maximum());
}
